Define templates¶

$classB(b)$ returns 9x9 constant background of height $b$

$classA(h, w, b)$ returns 9x9 2D-gaussian spot centered in the middle with the height $h$ and the width $w$ + background $b$

# template functions for classes A (spot) and B (no spot)
def classA(height, precision, background, D=9):
    loc = torch.ones(2) * (D-1)/2
    #precision = torch.ones(2) / width**2
    x, y = torch.meshgrid(torch.arange(D), torch.arange(D))
    pos = torch.stack((x, y), dim=2).float()
    rv = dist.MultivariateNormal(loc, precision_matrix=torch.eye(2)*precision)
    gaussian_spot = torch.exp(rv.expand([D,D]).log_prob(pos)) * 2 * np.pi / precision
    return height * gaussian_spot + background

def classB(background, D=9):
    return torch.ones(D,D)*background

def generate(height, width, background, noise, pi, D, K, N):
    # class templates
    locs = torch.zeros(K,D,D)
    locs[0,:,:] = classB(background, D)
    locs[1,:,:] = classA(height, width, background, D)
    with pyro.plate("sample_size", N):
        # hidden states; nospot p(0) = 0.4, spot p(1) = 0.6
        z = pyro.sample("z", dist.Categorical(pi))
        # add normal noise (std = 20) to template images
        data = pyro.sample("data", dist.Normal(locs[z,:,:], noise).to_event(2))
    return data, z

Bayesian analysis¶

Variables¶

$X = \{x_{n}\}$ (N images)

$ x_n = \{x_{n,d}\} $ (D pixels; D = 9x9 = 81)

$\pi = \{\pi_{k}\} = [\pi_0, \pi_1]$ (K classes; K = 2)

$\mu = \{ \mu_{k} \} = [\mu_{0}, \mu_{1}]$ (K classes)

$ \mu_k = \{\mu_{k,d}\} $ (D pixels)

$\mu_{0} = classB(b) $ (b - background level)

$\mu_{1} = classA(h, w, b) $ (h, w - height and width of the 2D Gaussian spot)

$\sigma$ is the noise in the image

Bayes' theorem¶

$$ p(\pi, \mu, \sigma | X) = \dfrac{p(X | \pi, \mu, \sigma) p(\pi, \mu, \sigma)}{p(X)} = \dfrac{p(X | \pi, \mu, \sigma) p(\pi) p(\mu) p(\sigma)}{p(X)} $$

Marginal likelihood of one image:

$$p (x_n|\pi, \mu, \sigma) = \sum_k^K p(z=k) p(x_n|\mu_k, \sigma) = \sum_{k}^{K} \pi_{k} \mathcal{N}(x_n|\mu_{k},\sigma) $$

where:

$$ \mathcal{N}(x_n|\mu_{k},\sigma) = \prod_d^D \mathcal{N}(x_{n,d}|\mu_{k,d},\sigma) $$

Marginal likelihood of N images:

$$p(X|\pi, \mu, \sigma) = \prod_{n}^{N} \bigg[ \sum_{k}^K \pi_{k} \mathcal{N}(x_{n}|\mu_{k},\sigma) \bigg]$$

Priors¶

$ p(\pi) = Dir(\alpha_0) $

$ p(\mu) = p(h, w, b) = p(h)p(w)p(b) $

$ p(h) = HalfNormal(500) $

$ p(w) = Gamma(1, 0.1) $

$ p(b) = HalfNormal(1000) $

$ p(\sigma) = Gamma(1, 0.001) $

Guides (approximations to posterior distributions)¶

$ p(\pi) = Dir(\pi\_conc) $

$ p(\mu) = p(h, w, b) = p(h)p(w)p(b) $

$ p(h) = \mathcal{N}(h\_loc, h\_scale) $

$ p(w^2) = InverseGamma(w\_alpha, w\_beta)$

$ p(b) = \mathcal{N}(b\_loc, b\_scale) $

$ p(\sigma^2) = InverseGamma(\sigma\_alpha, \sigma\_beta)$

@config_enumerate
def model(data, K):
    sample_size, N, _ = data.shape
    # prior distributions
    weights = pyro.sample("weights", dist.Dirichlet(0.5 * torch.ones(K)))
    height = pyro.sample("height", dist.HalfNormal(500.))
    #height = pyro.sample("height", dist.Uniform(0., 10000.))
    background = pyro.sample("background", dist.HalfNormal(1000.))
    #background = pyro.sample("background", dist.Uniform(0., 10000.))
    width = pyro.sample("width", dist.Gamma(1., 0.1)) # dist.HalfNormal(100.)
    # class templates
    locs = torch.zeros((K,N,N))
    locs[0,:,:] = classB(background)
    locs[1,:,:] = classA(height, width, background)
    sigma = pyro.sample("sigma", dist.Gamma(1, 0.001))
    with pyro.plate("sample_size", sample_size, device=device):
        # hidden states
        z = pyro.sample("z", dist.Categorical(weights))
        # likelihood / conditioning on data
        pyro.sample("obs", dist.Normal(locs[z,:,:], sigma).to_event(2), obs=data)
        
def guide(data, K):
    sample_size, N, _ = data.shape
    # posterior approximations
    weights_conc = pyro.param("weights_conc", torch.ones(2))
    weights = pyro.sample("weights", dist.Dirichlet(weights_conc))
    
    height_loc = pyro.param("height_loc", torch.tensor(100.))
    height_scale = pyro.param("height_scale", torch.tensor(20.), constraint=constraints.positive)
    height = pyro.sample("height", dist.Normal(height_loc, height_scale)) # true distribution
    
    background_loc = pyro.param("background_loc", torch.tensor(360.))
    background_scale = pyro.param("background_scale", torch.tensor(50.), constraint=constraints.positive)
    background = pyro.sample('background', dist.Normal(background_loc, background_scale))
    
    width_alpha = pyro.param("width_alpha", torch.tensor(10.), constraint=constraints.positive)
    width_beta = pyro.param("width_beta", torch.tensor(20.), constraint=constraints.positive)
    width = pyro.sample("width", dist.Gamma(width_alpha, width_beta))
    
    sigma_alpha = pyro.param("sigma_alpha", torch.tensor(20.), constraint=constraints.positive)
    sigma_beta = pyro.param("sigma_beta", torch.tensor(4.), constraint=constraints.positive)
    sigma = pyro.sample("sigma", dist.Gamma(sigma_alpha, sigma_beta))

@config_enumerate
def full_guide(data, K):
    sample_size, N, _ = data.shape
    # Global variables.
    with poutine.block(hide_types=["param"]):  # Keep our learned values of global parameters.
        guide(data, K)
    # Local variables.
    with pyro.plate("sample_size", sample_size):
        z_probs = pyro.param("z_probs", torch.ones(sample_size, K) / K, constraint=constraints.simplex)
        pyro.sample("z", dist.Categorical(z_probs))

def delta_guide(data, N, K, sample_size):
    auto_weights = pyro.param("auto_weights", torch.ones(2)*0.5, constraint=constraints.simplex)
    weights = pyro.sample("weights", dist.Delta(auto_weights, event_dim=1))
    auto_height = pyro.param("auto_height", torch.tensor(100.))
    height = pyro.sample('height', dist.Delta(auto_height)) # true distribution
    auto_background = pyro.param("auto_background", torch.tensor(360.))
    background = pyro.sample('background', dist.Delta(auto_background))
    return {"weights": weights, "height": height, "background": background}
